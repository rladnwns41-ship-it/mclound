어떤 언어든
몇 줄로 연결

Python · JavaScript · curl 무엇이든. HTTP 하나로 다 됩니다.

from aoaocloud_client import AOAOCloud

cloud = AOAOCloud("https://you.mcloud.dev", "aoao_...")

# SQL 그대로
cloud.s3.sql("CREATE TABLE items (id INTEGER, name TEXT)")

# 서버리스 함수
cloud.lambda_fn.create("hi", 'def h(e): return "Hi " + e["n"]')
r = cloud.lambda_fn.invoke("hi", {"n": "world"})
const API = "https://you.mcloud.dev"
const KEY = "aoao_anon_..."

// 리더보드 조회 - RLS 자동 적용
const r = await fetch(`${API}/api/s3/sql`, {
  method: "POST",
  headers: { "X-API-Key": KEY },
  body: JSON.stringify({
    query: "SELECT * FROM scores ORDER BY score DESC LIMIT 10"
  })
}).then(r => r.json())
# 회원가입 → API 키 발급
curl -X POST https://you.mcloud.dev/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{"username":"me","password":"strongpass1"}'

# SQL 실행
curl -X POST https://you.mcloud.dev/api/s3/sql \
  -H "X-API-Key: aoao_..." \
  -d '{"query":"SELECT * FROM users"}'
